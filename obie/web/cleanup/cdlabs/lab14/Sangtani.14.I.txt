#!/bin/bash
# Script: jsbackup
# Backs up files and puts the backed up files in the backup directory.
if [[ $# != 2 ]]
	then
		echo "use 2 arguments"
	else
		if [[ ! -r $1 ]]
			then
				echo "$1 not a file."
			else
				if [[ ! -d $2 ]]
				then
					echo "$2 not a directory."
				else
					for file in $(cat $1)
					do
						cp $file $2/$file.bak
					done
				fi
		fi
fi
